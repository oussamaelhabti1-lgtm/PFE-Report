	\chapter{Introduction et Objectifs}
	
	\section{Contexte de l'Évaluation}
	
	Ce rapport documente une évaluation de sécurité complète de l'application Damn Vulnerable Web Application (DVWA) v1.10, 
	déployée dans une infrastructure de laboratoire segmentée. L'objectif principal est de démontrer les vulnérabilités 
	présentes dans la liste OWASP Top 10: 2025 et de valider l'efficacité des techniques de détection et d'exploitation.
	
	L'évaluation a été réalisée dans un environnement contrôlé, incluant :
	\begin{itemize}
		\item Une architecture réseau segmentée (WAN/DMZ/LAN) protégée par un pare-feu pfSense
		\item Un serveur Web vulnérable (DVWA) déployé dans la DMZ via conteneurisation Docker
		\item Une machine attaquante (Parrot Security OS) simulant une menace externe
		\item Des systèmes de détection d'intrusion (Snort, pfBlockerNG)
	\end{itemize}
	
	\section{Objectifs de Sécurité}
	
	L'évaluation vise à atteindre les objectifs suivants :
	
	\begin{enumerate}
		\item \textbf{Identification :} Découvrir et documenter les vulnérabilités OWASP Top 10 présentes dans DVWA
		\item \textbf{Exploitation :} Démontrer pratiquement chaque vulnérabilité via des attaques contrôlées
		\item \textbf{Analyse :} Évaluer l'impact de chaque vulnérabilité sur la sécurité globale de l'application
		\item \textbf{Remédiation :} Proposer des solutions correctives pour éliminer ou atténuer les risques
	\end{enumerate}
	
	\section{Méthodologie}
	
	La démarche suivie respecte les standards de l'industrie :
	
	\begin{description}
		\item[Reconnaissance] : Identification des surfaces d'attaque exposées
		\item[Énumération] : Détection des services et versions
		\item[Exploitation] : Démonstration pratique de l'accès non autorisé
		\item[Post-Exploitation] : Analyse des conséquences de chaque compromission
		\item[Documentation] : Enregistrement détaillé avec preuves (screenshots)
	\end{description}

	\section{A05: Injection - SQL Injection}
	
	\subsection{Description de la Vulnérabilité}
	
	SQL Injection est une technique d'attaque qui consiste à injecter du code SQL malveillant 
	dans les paramètres d'une application pour manipuler les requêtes de la base de données.
	
	\subsection{Étapes d'Exploitation}
	
	\begin{enumerate}
		\item Accéder à : \textbf{Vulnerabilities → SQL Injection}
		\item Entrer la charge utile dans le champ \textit{User ID} :
		\begin{lstlisting}[caption=Payload SQL Injection,numbers=none,frame=none,backgroundcolor={},rulecolor={}]
			' OR '1'='1' --
		\end{lstlisting}
		\item Cliquer sur \textbf{Submit}
	\end{enumerate}
	
	\subsection{Résultats Observés}
	
	\textbf{Réponse de l'Application :}
	\begin{lstlisting}[caption=Message d'erreur révélateur]
		You have an error in your SQL syntax; check the manual that corresponds 
		to your MariaDB server version for the right syntax to use near ''' at line 1
	\end{lstlisting}
	
	\textbf{Informations Divulguées :}
	\begin{itemize}
		\item Type de base de données: \textbf{MariaDB}
		\item Détails de la requête SQL malformée
		\item Structure de la requête vulnérable
	\end{itemize}
	
	\subsection{Impact de Sécurité}
	
	\begin{tcolorbox}[colback=criticalred!10!white, colframe=criticalred, 
		title=\textbf{CRITICITÉ: CRITIQUE}]
		\begin{itemize}
			\item \textbf{Confidentialité :} CRITIQUE - Accès à tous les enregistrements base de données
			\item \textbf{Intégrité :} CRITIQUE - Modification/suppression de données possibles
			\item \textbf{Disponibilité :} CRITIQUE - Suppression complète de la base de données
			\item \textbf{CVSS v3.1 :} 9.8 (Critique)
		\end{itemize}
	\end{tcolorbox}

	\subsection{Remédiation}
	
	\begin{lstlisting}[language=php,caption=Requête Sécurisée avec Prepared Statements]
		// VULNÉRABLE (À NE PAS UTILISER)
		$query = "SELECT * FROM users WHERE id = " . $_GET['id'];
		
		// SÉCURISÉ (Utiliser toujours les prepared statements)
		$stmt = $mysqli->prepare("SELECT * FROM users WHERE id = ?");
		$stmt->bind_param("i", $_GET['id']);
		$stmt->execute();
		$result = $stmt->get_result();
	\end{lstlisting}

	\section{A05: Injection - Command Injection}
	
	\subsection{Description}
	
	Command Injection permet d'exécuter des commandes système arbitraires en exploitant 
	un champ de saisie qui ne valide pas les d��limiteurs de commande.
	
	\subsection{Exploitation}
	
	\begin{enumerate}
		\item Accéder à : \textbf{Vulnerabilities → Command Injection}
		\item Entrer l'adresse IP: \texttt{8.8.8.8}
		\item Ajouter la commande à chaîner: \texttt{; id}
		\item Charge utile complète : \texttt{8.8.8.8; id}
	\end{enumerate}
	
	\subsection{Résultats}
	
	\begin{lstlisting}[caption=Résultat de Command Injection,style=logstyle]
		PING 8.8.8.8 (8.8.8.8): 56 data bytes
		64 bytes from 8.8.8.8: icmp_seq=0 ttl=126 time=353.420 ms
		64 bytes from 8.8.8.8: icmp_seq=1 ttl=126 time=24.713 ms
		64 bytes from 8.8.8.8: icmp_seq=2 ttl=126 time=22.696 ms
		64 bytes from 8.8.8.8: icmp_seq=3 ttl=126 time=23.053 ms
		--- 8.8.8.8 ping statistics ---
		4 packets transmitted, 4 packets received, 0% packet loss
		
		uid=33(www-data) gid=33(www-data) groups=33(www-data)
	\end{lstlisting}
	
	\subsection{Impact}
	
	\begin{tcolorbox}[colback=criticalred!10!white, colframe=criticalred, 
		title=\textbf{CRITICITÉ: CRITIQUE}]
		\textbf{Accès Système Complet :} Attaquant exécute des commandes avec les 
		privilèges du serveur web (www-data). Possibilité complète de compromission 
		du système (installation de backdoors, vol de données, DoS).
	\end{tcolorbox}
	
	\subsection{Screenshot}
	
	\begin{figure}[H]
		\centering
		\includegraphics[width=0.95\textwidth]{command-injection.png}
		\caption{Command Injection - Exécution de Commandes Système}
		\label{fig:command-injection}
	\end{figure}
	
	\subsection{Remédiation}
	
	\begin{lstlisting}[language=php,caption=Protection contre Command Injection]
		// VULNÉRABLE
		exec("ping -c 4 " . $_GET['ip']);
		
		// SÉCURISÉ - Utiliser escapeshellarg()
		$ip = escapeshellarg($_GET['ip']);
		exec("ping -c 4 " . $ip);
		
		// MIEUX ENCORE - Whitelist
		$allowed_ips = array('8.8.8.8', '8.8.4.4');
		if (in_array($_GET['ip'], $allowed_ips)) {
			exec("ping -c 4 " . escapeshellarg($_GET['ip']));
		}

	\section{A07: Authentication Failures - Brute Force Login}
	
	\subsection{Description}
	
	L'absence de mécanisme de verrouillage de compte ou de limitation de débit 
	permet les attaques par force brute illimitées contre les identifiants.
	
	\subsection{Exploitation}

		function log_failed_attempt($username, $ip) {
			// Incrémenter compteur dans Redis/Memcache
			$key = "failed_login:{$username}:{$ip}";
			$redis->incr($key);
			$redis->expire($key, 900); // 15 minutes
			
			// Vérifier le seuil

		usleep(($failed_attempts ^ 2) * 1000000); // 1, 4, 9... secondes

	\subsection{Remédiation}
	
	\begin{lstlisting}[language=php,caption=Upload Sécurisé]
		// Whitelist d'extensions autorisées
		$allowed = array('jpg', 'jpeg', 'png', 'gif', 'pdf');
		$filename = basename($_FILES['file']['name']);
		$ext = strtolower(pathinfo($filename, PATHINFO_EXTENSION));
		
		if (!in_array($ext, $allowed)) {
			die("Extension de fichier non autorisée!");
		}
		
		// Vérifier le type MIME
		$finfo = finfo_open(FILEINFO_MIME_TYPE);
		$mime = finfo_file($finfo, $_FILES['file']['tmp_name']);
		$allowed_mimes = array('image/jpeg', 'image/png', 'image/gif', 'application/pdf');
		
		if (!in_array($mime, $allowed_mimes)) {
			die("Type MIME non autorisé!");
		}
		
		// Stocker EN DEHORS du répertoire web
		$upload_dir = '/var/uploads/secure/'; // Hors de /var/www/html
		move_uploaded_file($_FILES['file']['tmp_name'], 
		$upload_dir . uniqid() . '.' . $ext);
	\end{lstlisting}
